302  void slotKeyPressed(void)
303  {
304  QToolButton *button = qobject_cast<QToolButton *> (sender());
305 
306  if(!button)
307  return;
308 
309  QString text(m_ui.passphrase->text());
310 
311  if(button == m_ui.space)
312  text.append(" ");
313  else if(m_ui.shift->isChecked())
314  text.append(button->text().split('\n').value(0).mid(0, 1).toUpper());
315  else if(button->text().contains('\n'))
316  text.append(button->text().split('\n').value(1));
317  else
318  text.append(button->text());
319 
320  m_ui.passphrase->setText(text);
321  }
